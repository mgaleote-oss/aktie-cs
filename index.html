// ============================================================ // Coaktion CS · Apps Script completo v2 // Funções: // 1. salvarNoLog() → botão da Calculator (sem alteração) // 2. doGet() → API do dashboard (health scores + // notes + tasks + docs + ai_recs) // ============================================================ const SECRET_TOKEN = 'hs_minha_chave_2026'; // ← troque e anote // ── 1. SALVAR NO LOG ──────────────────────────────────────── function salvarNoLog() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var calc = ss.getSheetByName('Calculator'); var db = ss.getSheetByName('Database'); var conta = calc.getRange('B7').getValue(); var data = calc.getRange('H7').getValue(); if (!conta || conta === 'Nome da Conta') { SpreadsheetApp.getUi().alert('Preencha o nome da conta antes de salvar.'); return; } if (!data) { SpreadsheetApp.getUi().alert('Preencha a data antes de salvar.'); return; } var account = calc.getRange('B7').getValue(); var tipo = calc.getRange('D7').getValue(); var periodo = calc.getRange('F7').getValue(); var dataVal = calc.getRange('H7').getValue(); var csm = calc.getRange('I7').getValue(); var scores = calc.getRange('D11:D20').getValues(); var scoreVal = calc.getRange('E22').getValue(); var status = scoreVal >= 8 ? 'SAUDAVEL' : scoreVal >= 6 ? 'ATENCAO' : 'RISCO'; var lastRow = db.getLastRow(); var nextRow = Math.max(lastRow + 1, 4); var rowData = [ account, tipo, periodo, dataVal, csm, scores[0][0], scores[1][0], scores[2][0], scores[3][0], scores[4][0], scores[5][0], scores[6][0], scores[7][0], scores[8][0], scores[9][0], scoreVal, status ]; db.getRange(nextRow, 2, 1, rowData.length).setValues([rowData]); db.getRange(nextRow, 5).setNumberFormat('DD/MM/YYYY'); var statusCell = db.getRange(nextRow, 18); if (status === 'SAUDAVEL') statusCell.setFontColor('#1E8449'); else if (status === 'ATENCAO') statusCell.setFontColor('#7D6608'); else statusCell.setFontColor('#922B21'); statusCell.setFontWeight('bold'); calc.getRange('B7').setValue(''); calc.getRange('D7').setValue('Projects'); calc.getRange('F7').setValue('14 dias'); calc.getRange('H7').setValue(''); calc.getRange('I7').setValue(''); calc.getRange('D11:D20').setValue(5); SpreadsheetApp.getUi().alert( 'Salvo na linha ' + nextRow + '!\nConta: ' + account + ' | Score: ' + scoreVal.toFixed(1) ); ss.setActiveSheet(calc); } // ── 2. API DO DASHBOARD ───────────────────────────────────── function doGet(e) { if (!e || !e.parameter || e.parameter.token !== SECRET_TOKEN) { return jsonOut({ error: 'Unauthorized' }); } var action = e.parameter.action || 'read'; // ── WRITE ACTIONS ── if (action === 'saveAI') { return saveAIRec(e.parameter); } if (action === 'note_add') { return saveNote(e.parameter); } if (action === 'task_add') { return saveTask(e.parameter, 'add'); } if (action === 'task_toggle') { return saveTask(e.parameter, 'toggle'); } if (action === 'task_delete') { return saveTask(e.parameter, 'delete'); } // ── READ ALL ── var ss = SpreadsheetApp.getActiveSpreadsheet(); return jsonOut({ ok: true, data: readDatabase(ss), notes: readNotes(ss), tasks: readTasks(ss), docs: readDocs(ss), ai_recs: readAIRecs(ss), }); } // ── READ: Database ─────────────────────────────────────────── function readDatabase(ss) { var db = ss.getSheetByName('Database'); var rows = db.getDataRange().getValues(); var headerRow = -1; for (var i = 0; i < rows.length; i++) { var cell = String(rows[i][1] || '').toLowerCase(); if (cell.indexOf('conta') !== -1 || cell.indexOf('account') !== -1) { headerRow = i; break; } } if (headerRow === -1) return []; var data = []; for (var i = headerRow + 1; i < rows.length; i++) { var r = rows[i]; var account = String(r[1] || '').trim(); if (!account) continue; var dataVal = r[4]; if (dataVal instanceof Date) { dataVal = pad(dataVal.getDate()) + '/' + pad(dataVal.getMonth()+1) + '/' + dataVal.getFullYear(); } else { dataVal = String(dataVal || '').trim(); } data.push({ account: account, tipo: String(r[2] || '').trim(), periodo: String(r[3] || '').trim(), data: dataVal, csm: String(r[5] || '').trim(), scores: [toNum(r[6]),toNum(r[7]),toNum(r[8]),toNum(r[9]),toNum(r[10]), toNum(r[11]),toNum(r[12]),toNum(r[13]),toNum(r[14]),toNum(r[15])], score_final: toNum(r[16]), status: String(r[17] || '').trim() }); } return data; } // ── READ: Notes (Google Forms responses) ──────────────────── // Aba esperada: "Notes" com colunas Timestamp | Conta | Anotação function readNotes(ss) { var sheet = ss.getSheetByName('Notes'); if (!sheet) return []; var rows = sheet.getDataRange().getValues(); var notes = []; for (var i = 1; i < rows.length; i++) { // pula cabeçalho var r = rows[i]; var account = String(r[1] || '').trim(); var texto = String(r[2] || '').trim(); if (!account || !texto) continue; var dataVal = r[0]; // Timestamp do Forms if (dataVal instanceof Date) { dataVal = pad(dataVal.getDate()) + '/' + pad(dataVal.getMonth()+1) + '/' + dataVal.getFullYear(); } else { dataVal = String(dataVal || '').trim(); } notes.push({ account: account, data: dataVal, texto: texto }); } return notes; } // ── READ: Tasks ────────────────────────────────────────────── // Aba: "Tasks" com colunas ID | Descrição | Cliente | Data | Done function readTasks(ss) { var sheet = ss.getSheetByName('Tasks'); if (!sheet) return []; var rows = sheet.getDataRange().getValues(); var tasks = []; for (var i = 1; i < rows.length; i++) { var r = rows[i]; var id = String(r[0] || '').trim(); if (!id) continue; tasks.push({ id: id, desc: String(r[1] || '').trim(), client: String(r[2] || '').trim(), date: String(r[3] || '').trim(), done: r[4] === true || String(r[4]).toLowerCase() === 'true' }); } return tasks; } // ── READ: Documents ────────────────────────────────────────── // Aba: "Documents" com colunas Nome | Link | Categoria function readDocs(ss) { var sheet = ss.getSheetByName('Documents'); if (!sheet) return []; var rows = sheet.getDataRange().getValues(); var docs = []; for (var i = 1; i < rows.length; i++) { var r = rows[i]; var nome = String(r[0] || '').trim(); if (!nome) continue; docs.push({ nome: nome, link: String(r[1] || '').trim(), categoria: String(r[2] || '').trim() || 'Outro' }); } return docs; } // ── READ: AI Recommendations ───────────────────────────────── // Aba: "AI_Recommendations" com colunas Conta | Data | Recomendação function readAIRecs(ss) { var sheet = ss.getSheetByName('AI_Recommendations'); if (!sheet) return {}; var rows = sheet.getDataRange().getValues(); var recs = {}; for (var i = 1; i < rows.length; i++) { var r = rows[i]; var account = String(r[0] || '').trim(); if (!account) continue; recs[account] = { date: String(r[1] || '').trim(), text: String(r[2] || '').trim() }; } return recs; } // ── WRITE: Note ────────────────────────────────────────────── function saveNote(params) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var account = params.account || ''; var date = params.date || ''; var texto = params.texto || ''; if (!account || !texto) return jsonOut({ error: 'Missing fields' }); var sheet = ss.getSheetByName('Notes'); if (!sheet) { sheet = ss.insertSheet('Notes'); sheet.getRange(1,1,1,3).setValues([['Timestamp','Conta','Anotação']]); } sheet.appendRow([date, account, texto]); return jsonOut({ ok: true }); } // ── WRITE: AI Recommendation ───────────────────────────────── function saveAIRec(params) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var account = params.account || ''; var date = params.date || ''; var text = params.text || ''; if (!account) return jsonOut({ error: 'Missing account' }); var sheet = ss.getSheetByName('AI_Recommendations'); if (!sheet) { sheet = ss.insertSheet('AI_Recommendations'); sheet.getRange(1,1,1,3).setValues([['Conta','Data','Recomendação']]); } // Atualiza se já existe, senão adiciona var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]).trim() === account) { sheet.getRange(i+1, 1, 1, 3).setValues([[account, date, text]]); return jsonOut({ ok: true }); } } sheet.appendRow([account, date, text]); return jsonOut({ ok: true }); } // ── WRITE: Tasks ───────────────────────────────────────────── function saveTask(params, action) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Tasks'); if (!sheet) { sheet = ss.insertSheet('Tasks'); sheet.getRange(1,1,1,5).setValues([['ID','Descrição','Cliente','Data','Concluída']]); } var id = params.id || ''; if (action === 'add') { sheet.appendRow([ id, params.desc || '', params.client || '', params.date || '', false ]); return jsonOut({ ok: true }); } var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]).trim() === id) { if (action === 'toggle') { sheet.getRange(i+1, 5).setValue(params.done === 'true'); } else if (action === 'delete') { sheet.deleteRow(i+1); } return jsonOut({ ok: true }); } } return jsonOut({ error: 'Task not found' }); } // ── Helpers ────────────────────────────────────────────────── function toNum(val) { if (val === null || val === undefined || val === '') return 5; return parseFloat(String(val).replace(',', '.')) || 5; } function pad(n) { return n < 10 ? '0' + n : String(n); } function jsonOut(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ============================================================ // ABAS NECESSÁRIAS NO SHEETS: // - Database (já existe) // - Notes (criada automaticamente pelo script) // - Tasks (criada automaticamente pelo script) // - Documents (crie com: Nome | Link | Categoria) // - AI_Recommendations (criada automaticamente pelo script) // // IMPORTANTE: após adicionar este código, faça uma // NOVA implantação (não editar a existente) para que // as novas rotas da API funcionem. // ============================================================
